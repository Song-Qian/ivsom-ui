基础用法

常规计数器

  • InputNumber 组件在使用之前，需要有相关的依赖库进行支持，如需阅读：快速开始
  • max : number 最大值
  • min : number 最小值
  • step : float | number 每次值跳跃的跨度
  • size: 'medium' | 'small' | 'mini' 表单元素大小
  • disabled: Boolean 是否禁用
  • precision: number 数值精度
  • name: 原生表单属性
  • input: Boolean 是否显示输入框
<i-vsom-number v-model="value" ></i-vsom-number>   

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            value: 0
        }
    }
 })

Input Number Event 事件

  • onChange: Arguments(e : Event, value : string) 值改变的事件。
  • onLess: Arguments(e : MouseEvent) 值递减按钮点击事件
  • onPlus: Arguments(e : MouseEvent) 值递增按钮点击事件

值改变的事件

<i-vsom-number v-model="value" :step="0.1" :precision="2" size="small" @on-change="onChange"></i-vsom-number>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            value : 0
        }
    },
    methods : {
        onChange : function(e, value) {
            alert(value)
        }
    }
})

值递减、递增按钮点击事件

<i-vsom-number v-model="value" :step="0.1" :precision="2" size="small" @on-less="onLess" @on-plus="onPlus"></i-vsom-number>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            value : 0
        }
    },
    methods : {
        onLess : function(e) {
            alert("递减");
        },
        onPlus : function(e) {
            alert("递增");
        }
    }
})
