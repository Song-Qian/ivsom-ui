复选框组的表单验证

苹果 草莓 香蕉 火龙果 橙子 榴莲 百香果 点击验证 
<i-vsom-form :model="form" :label-width="80" position="left">
    <i-vsom-form-validater label="水果" prop="fruit" trigger="change" :validate="validateForm" error-message="当前选中的水果已售完" required>
        <i-vsom-check-box-group name="fruit" v-model="form.fruit">
            <i-vsom-check-box size="small" value="苹果">苹果</i-vsom-check-box>
            <i-vsom-check-box size="small" value="草莓">草莓</i-vsom-check-box>
            <i-vsom-check-box size="small" value="香蕉">香蕉</i-vsom-check-box>
            <i-vsom-check-box size="small" value="火龙果">火龙果</i-vsom-check-box>
            <i-vsom-check-box size="small" value="橙子">橙子</i-vsom-check-box>
            <i-vsom-check-box size="small" value="榴莲">榴莲</i-vsom-check-box>
            <i-vsom-check-box size="small" value="百香果">百香果</i-vsom-check-box>
        </i-vsom-check-box-group>
    </i-vsom-form-validater>
</i-vsom-form>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            form: {
                fruit : []
            }
        }
    },
    methods : {
        validateForm : function(callback) {
            let me = this;
            if(me.form.fruit.indexOf('香蕉') > -1) {
                return callback(new Error('香蕉未进货。'))
            }

            if(me.form.fruit.indexOf('百香果') > -1) {
                return callback(new Error('当前选中的水果已售完'))
            }
            return callback()
        }
    }
})
