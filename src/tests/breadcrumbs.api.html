                { text : '面包屑二', url : '#2'},

                { text : '面包屑七', url : '#7'}
            ]
        }
    },
    methods : {
    }
})

自定义符号

  • Breadcrumbs 组件支持使用ICON 组件当成面包屑中的分隔符号
