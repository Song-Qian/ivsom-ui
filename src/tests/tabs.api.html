基础用法

默认样式

  • Tabs 组件在使用之前，需要有相关的依赖库进行支持，如需阅读：快速开始
  • Tabs 组件具有默认的宽度 (100%) 和高度 (auto), 且Tab Nav栏的高度固定占用40px
  • iVsomTabs 与 iVsomTabsPanel 需配合使用，单独使用任一组件，将会无法达到预定期望。
  • Tabs Props:
  • width: number | string 宽度
  • height: number | string Tabs 高度
  • value: string Tabs Pages 被选中的值
  • flex: boolean Tab Panel 是否响应式容器大小
  • stretch: boolean Tab Label 是否自撑开宽度
  • type: 'default' | 'card' | 'nav' Tabs 外形样式定义
  • closable: boolean Tabs Nav 关闭按钮
  • opne: boolean Tabs Nav 新增按钮
  • TabPanel Props:
  • name: string Tabs label 标题
  • index: string Tab label 选中值

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            index : '1',
            index_1 : '0',
            tabs : [
                { name : "Tabs - 1", index : "1" },
                { name : "Tabs - 2", index : "2" },
                { name : "Tabs - 3", index : "3" },
                { name : "Tabs - 4", index : "4" }
            ],
            navs : {
                '0' : 'Tabs - 0',
                '1' : 'Tabs - 1',
                '2' : 'Tabs - 2',
                '3' : 'Tabs - 3',
                '4' : 'Tabs - 4',
                '5' : 'Tabs - 5',
                '6' : 'Tabs - 6',
                '7' : 'Tabs - 7',
                '8' : 'Tabs - 8',
                '9' : 'Tabs - 9',
                '10' : 'Tabs - 10',
                '11' : 'Tabs - 11',
                '12' : 'Tabs - 12',
                '13' : 'Tabs - 13',
                '14' : 'Tabs - 14',
                '15' : 'Tabs - 15',
                '16' : 'Tabs - 16',
                '17' : 'Tabs - 17',
            }
        }
    },
    methods : {
        onTabsNavClose : function(panes) {
            var me = this;
            var index = panes.index;
            me.tabs = me.tabs.filter(function(item) { return item.index !== index });
            if(me.index === index) {
                me.index = me.tabs[0].index;
            }
        },
        onTabsNavClose_1 : function(panes) {
            var me = this;
            var index = panes.index;
            me.$delete(me.navs, index);
            if(me.index_10 === index) {
                me.index_10 = Object.keys(me.navs)[0];
            }
        }
    }
})
