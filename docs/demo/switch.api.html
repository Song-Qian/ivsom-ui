基础用法

常规switch

  • switch 组件在使用之前，需要有相关的依赖库进行支持，如需阅读：快速开始
  • name : string 表单字段名称
  • checked : Boolean 状态开关默认值
  • size: 'medium' | 'small' | 'mini' 表单元素大小
  • disabled: Boolean 是否禁用
  • active-color: #3bd134 激活开关状态颜色
  • inactive-color: #f75f3c 失活开关状态颜色
<i-vsom-switch v-model="value"></i-vsom-switch>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            value: true
        }
    }
})

var app = new Vue({

开关的表单验证

<i-vsom-form :model="form" :label-width="80" position="left">
    <i-vsom-form-validater label="扩音器" prop="open" trigger="change" :validate="(cb) => form.open ? cb() : cb(new Error())" error-message="当前已关闭扩音器">
        <i-vsom-switch v-model="form.open" size="small"></i-vsom-switch>
    </i-vsom-form-validater>
</i-vsom-form>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            form : {
                open: false
            }
        }
    }
})

Switch Event 事件

  • onChange: Arguments(e: MouseEvent, value: Boolean) Switch 引发值改变的事件。
<i-vsom-switch v-model="value" size="small" @on-change="onChange" ></i-vsom-switch>

var app = new Vue({
    el : '#app',
    data : function() {
        return {
            value : true
        }
    },
    methods : {
        onChange : function(e, value) {
            console.log(value);
        }
    }
})
